    for ln in fin:
          pid, sid, dist, pval, slope = ln.split()
          chrom, pos= sid.split(":")
          name=sid

  print_help(opt_parser)
  stop("Need input file", call.=FALSE)
}

  print_help(opt_parser)
  stop("Need output file", call.=FALSE)
